CONFIGURATION
    CARD_VALUE_OVERRIDE
        ace = 11;
        king = 10;
        queen = 10;
        joker = FALSE;
        roles = DEALER and PLAYER;
    END CARD_VALUE_OVERRIDE
    ACTIONS
        set HIT does pickCard();
        set HITHIDE does pickCard() and hideCard();
    END ACTIONS
    VARIABLES
        set DEALER has score;
        set PLAYER has score;
    END VARIABLES
